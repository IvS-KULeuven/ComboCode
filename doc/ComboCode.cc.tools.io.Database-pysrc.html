 721       
 722      '''

 724

 728      @keyword id_index: index of model id in the database entries, specify only

 733                         (default: None)

 742

 751      if id_index <> None and code <> None:

 759          else:

 761                  if k.lower() in os.path.split(db_path)[1].lower(): 
 762                      if id_index <> None:

 766                          raise ValueError(this_str) 
 767                      id_index = v 
 768                      code = k

 770          this_str2 = 'Cannot figure out which code the database is used for. '+\

 772          raise IOError(this_str2)
