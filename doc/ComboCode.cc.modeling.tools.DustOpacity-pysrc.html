 54      if not type(acut) is types.ListType: 
 55          acut = [float(acut)] 
 56      acut = sorted(acut) 
 57      #-- Slope is negative typically. If slope == -4, this doesn't work 
 58      if slope == -4.: return np.empty(0) 
 59      #-- n(a) = Cst * Int(a**3 * a**-3.5 da) ==> Cst * [a**0.5]^amax_amin 
 60      p = (3.+slope)+1. 
 61      subsets = [] 
 62      #-- first subset: 
 63      subsets.append(acut[0]**p-amin**p)     
 64      #-- additional subsets if len(acut) > 1 
 65      for i in xrange(len(acut[1:])): 
 66          subsets.append(acut[i+1]**p-acut[i]**p)

 68      subsets.append(amax**p-acut[-1]**p)
